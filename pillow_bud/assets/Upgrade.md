### 5.1 RANDOM MOTIVATIONAL QUOTE BOX

**Feature Overview:**
```
Location:           Home screen, top priority (after greeting)
Display:            Card-based glass container
Content:            10,000+ pre-written quotes
Update Frequency:   
  - Every app launch (new quote)
  - Refresh button for manual refresh
  - Daily collection (shows different quotes each day)
Space Used:         4-5 lines of text + author (120-160px height)
```

**Quote Structure (Database):**
```dart
class Quote {
  String id;              // Unique ID
  String text;            // The quote (max 200 chars)
  String author;          // Who said it
```

```dart
  static Quote getDailyQuote() {
    final now = DateTime.now();
    final seed = now.year * 10000 + now.month * 100 + now.day;
    final index = seed % quotesDatabase.length;
    return Quote.fromJson(quotesDatabase[index]);
  }

  static Quote getRandomQuote() {
    final random = DateTime.now().millisecondsSinceEpoch % quotesDatabase.length;
    return Quote.fromJson(quotesDatabase[random]);
  }

  static Quote getQuoteByCategory(String category) {
    final filtered = quotesDatabase
        .where((q) => q['category'] == category)
        .toList();
    
    if (filtered.isEmpty) return getDailyQuote();
    
    final index = DateTime.now().millisecondsSinceEpoch % filtered.length;
    return Quote.fromJson(filtered[index]);
  }

  static List<Quote> getTopQuotes({int limit = 10}) {
    return quotesDatabase
        .where((q) => q['isPopular'] == true)
        .take(limit)
        .map((q) => Quote.fromJson(q))
```

```dart
class BreathingGuide extends StatefulWidget {

class _BreathingGuideState extends State<BreathingGuide>

    _setupAnimations();

    _circleController = AnimationController(

      vsync: this,
    );

    _textController = AnimationController(
      duration: const Duration(milliseconds: 150),
      vsync: this,
    );
```

```dart
import 'package:flutter/material.dart';
import 'package:flutter_animate/flutter_animate.dart';
import 'package:pillow_bud/services/quote_service.dart';

class DailyQuoteWidget extends StatefulWidget {
  const DailyQuoteWidget({Key? key}) : super(key: key);

  @override
  State<DailyQuoteWidget> createState() => _DailyQuoteWidgetState();
}

class _DailyQuoteWidgetState extends State<DailyQuoteWidget>
    with TickerProviderStateMixin {
  late AnimationController _slideController;
  late Quote _currentQuote;

  @override
  void initState() {
    super.initState();
    _slideController = AnimationController(
      duration: Duration(milliseconds: 300),
      vsync: this,
    );
    _currentQuote = QuoteService.getDailyQuote();
  }

  void _refreshQuote() {
    _slideController.forward().then((_) {
      setState(() => _currentQuote = QuoteService.getRandomQuote());
      _slideController.reverse();
    });
  }

  @override
  Widget build(BuildContext context) {
    return SlideTransition(
      position: Tween<Offset>(
        begin: Offset(0, 0),
        end: Offset(-0.2, 0),
      ).animate(_slideController),
      child: Container(
        padding: EdgeInsets.all(16),
        decoration: BoxDecoration(
          color: Color(0xFF87CEEB).withOpacity(0.1),
          border: Border.all(
            color: Color(0xFF87CEEB).withOpacity(0.2),
            width: 1,
          ),
          borderRadius: BorderRadius.circular(16),
          backdropFilter: ImageFilter.blur(sigmaX: 8, sigmaY: 8),
        ),
        child: Column(
          children: [
            // Quote Text
            Text(
              _currentQuote.text,
              style: TextStyle(
                fontFamily: 'Poppins',
                fontSize: 16,
                fontWeight: FontWeight.w600,
                color: Colors.white,
                height: 1.5,
              ),
              textAlign: TextAlign.center,
            ),
            SizedBox(height: 12),

            // Author
            Text(
              '— ${_currentQuote.author}',
              style: TextStyle(
                fontFamily: 'Inter',
                fontSize: 12,
                fontWeight: FontWeight.w500,
                color: Color(0xFF87CEEB).withOpacity(0.8),
              ),
            ),
            SizedBox(height: 12),

            // Action Buttons
            Row(
              mainAxisAlignment: MainAxisAlignment.spaceEvenly,
              children: [
                IconButton(
                  onPressed: () {
                    // Save to favorites
                  },
                  icon: Icon(Icons.favorite_border, color: Colors.white),
                ),
                IconButton(
                  onPressed: _refreshQuote,
                  icon: Icon(Icons.refresh, color: Colors.white),
                ),
                IconButton(
                  onPressed: () {
                    // Share quote
                  },
                  icon: Icon(Icons.share, color: Colors.white),
                ),
              ],
            ),
          ],
        ),
      ),
    );
  }

  @override
  void dispose() {
    _slideController.dispose();
    super.dispose();
  }
}
```
